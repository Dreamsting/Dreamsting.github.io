    <li>1</li>

    <li>1</li>

html.xpath("//li[@title]")
9.2 获取倒数第二个值为2的HTNL节点
//li[@title][last()-1]/text()

title_list = html.xpath("//li[@title]")
for title in title_list:
    title_value = title.xpath("@title")

    print(title_value)
